ZIP Code Data Map

// Initialize Leaflet map var map = L.map('map').setView([37.8, -96], 4); // Add tile layer L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '© OpenStreetMap contributors' }).addTo(map); // Load data from Flask route L.geoJson.ajax("/zip_data", { onEachFeature: function (feature, layer) { layer.bindPopup( `ZIP Code: ${feature.properties.Zip}
` + `Joint: ${feature.properties.Joint}
` + `Single: ${feature.properties.Single}
` + `Head of Household: ${feature.properties.Hd_of_Hshld}` ); }, pointToLayer: function (feature, latlng) { return L.circleMarker(latlng); } }).addTo(map); // Log data to console for debugging fetch('/zip_data') .then(response => response.json()) .then(data => console.log(data)) .catch(error => console.error('Error fetching data:', error));
